#!/usr/bin/python3
"""Python program to divide values in a matrix"""

def matrix_divided(matrix, div):
    """ Function to divide values in a matrix

    Args:
        matrix (list of lists): a list of lists containing integers or floats
        div (int or float): divider

    Returns:
        A new matrix with values divided by div, rounded to 2 decimal places.
    """
    string = "matrix must be a matrix (list of lists) of integers/floats"
    
    # Check if div is a number
    if type(div) not in [int, float]:
        raise TypeError("div must be a number")
    
    # Check if matrix is a list of lists
    if not isinstance(matrix, list) or not all(isinstance(row, list) for row in matrix):
        raise TypeError(string)
    
    # Check if all elements in the matrix are either integers or floats
    if not all(isinstance(val, (int, float)) for row in matrix for val in row):
        raise TypeError(string)
    
    # Check if all rows have the same length
    row_length = len(matrix[0])
    if not all(len(row) == row_length for row in matrix):
        raise TypeError("Each row of the matrix must have the same size")
    
    # Perform division and round the result to 2 decimal places
    div_matrix = [[round(value / div, 2) for value in row] for row in matrix]

    return div_matrix
